Sample Workflow Code

import { TriggerDev } from "@trigger.dev/sdk";

const client = new TriggerDev({
  id: "github-to-slack",
  name: "GitHub to Slack Automation"
});

client.defineJob({
  id: "github-push",
  name: "GitHub Push Handler",
  version: "1.0.0",
  trigger: github.triggers.push({
    repo: "your-org/your-repo"
  }),
  run: async (payload, io, ctx) => {
    // AI-powered code analysis
    const analysis = await io.openai.chat("analyze", {
      model: "gpt-4",
      messages: [
        {
          role: "user", 
          content: `Analyze this commit: ${payload.commits[0].message}`
        }
      ]
    });

    // Send notification to Slack
    await io.slack.postMessage("notify", {
      channel: "#dev-updates",
      text: `üöÄ New commit analyzed: ${analysis.content}`
    });
  }
});
